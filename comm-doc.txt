
Http Endpoints

GET /v1/auth-ticket/:

  Header:
    {
      Content-Type: string,
      api-key: string
    }

  Request Body:
    None

  Response Success Body:
    {
      authKey: string
    }

Web-socket Messages

  # map generated id of courier client to redis of { id, location }
  # a single connection runs in one go-routine

  Inbound Messages:

    Queue Order (from consumer):
      {
      "orderInfo": {
          "location":{
            "lon": float,
            "lat": float
          },
          "description": string,
          "consumerId": string,
          "bidPrice": float,
          "duration": "string" (not sure if consumer should send?)
        }
      }

    Creating Connection Header:
      {
        authKey: string,
        courierId: string  # probably will generate id server side though, mapping
      }

    Accept Trip Payload:
      {
        message: accept order
      }

    Reject Trip Payload:
      {
        message: reject order
      }

  Outbound Messages:

    Offer Trip Payload:
      {
        event: order offer,
        data: {
          orderInfo: {
            id: string,
            description: string,
            duration: float
            bidPrice: float,
            timeRequested: string,
            location: {
              lon: float,
              lat: float
            }
          }
        }
      }

    Trip Match Confirmation:
      {
        event: order confirmed,
        data: {
          orderId: string
        }
      }

    Error Occurred:
      {
        event: error,
        data: {
          message: string,
          details: string
        }
      }
